function getLocation() {
    if (!navigator.geolocation) {
        document.getElementById('lokasiInfo').textContent = '⚠️ GPS tidak didukung.';
        return;
    }
    navigator.geolocation.getCurrentPosition(
        pos => {
            latitude = pos.coords.latitude;
            longitude = pos.coords.longitude;
            document.getElementById('lokasiInfo').innerHTML =
                '📍 ' + latitude.toFixed(6) + ', ' + longitude.toFixed(6) +
                ' (akurasi ' + Math.round(pos.coords.accuracy) + 'm)';
        },
        err => document.getElementById('lokasiInfo').textContent = '⚠️ Gagal GPS: ' + err.message,
        { enableHighAccuracy: true, timeout: 10000 }
    );
}

    var video = document.getElementById('video');
    var canvas = document.getElementById('canvas');
    var preview = document.getElementById('preview');
    var fotoSelfie = document.getElementById('fotoSelfie');

    if (stream && stream.active) { ambilGambar(); return; }

    if (stream) { stream.getTracks().forEach(t => t.stop()); stream = null; }

            video.srcObject = s;
            video.play();

            let msg = '⚠️ Gagal kamera: ';

# ROUTE TERIMA ABSENSI

@app.route('/absen', methods=['POST'])
def absen():
    try:
        data = request.get_json()
        nama = data.get('nama')
        kategori = data.get('kategori')
        lat = data.get('latitude')
        lon = data.get('longitude')
        foto_base64 = data.get('foto')

        now = datetime.datetime.now()
        timestamp_str = now.strftime("%Y-%m-%d %H:%M:%S")
        filename = f"{now.strftime('%Y%m%d_%H%M%S')}_{nama}.jpg"

        # Simpan foto
        with open(filename, 'wb') as f:
            f.write(base64.b64decode(foto_base64.split(',')[1]))

        # Simpan ke CSV
        csv_file = 'absensi.csv'
        file_exists = os.path.isfile(csv_file)
        with open(csv_file, 'a', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)
            if not file_exists:
                writer.writerow(['Waktu', 'Nama', 'Kategori', 'Latitude', 'Longitude', 'Foto'])
            writer.writerow([timestamp_str, nama, kategori, lat, lon, filename])

        return jsonify({
            'success': True,
            'message': 'Absen berhasil disimpan!',
            'waktu': timestamp_str
        })

    except Exception as e:
        return jsonify({
            'success': False,
            'message': str(e)
        }), 500
